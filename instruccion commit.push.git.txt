# Comandos de Git para commit y push desde el directorio Inandes.TECH
# Ejecutar estos comandos desde el directorio C:\Users\rguti\

# 1. Asegurarse de estar en el directorio correcto y añadir solo los archivos .py
#    (Este comando asume que ya estás en C:\Users\rguti\)
cd Inandes.TECH && git add *.py

# 2. Realizar el commit con el mensaje especificado
cd Inandes.TECH && git commit -m "Refactorización de calculadora a agnóstica y actualización de UI - 20.08 11 PM"

# 3. Realizar el push forzado al repositorio remoto
cd Inandes.TECH && git push --force

# NOTA: Los comandos de 'git init', 'git remote add' y '.gitignore'
fueron omitidos ya que el repositorio ya existe y está configurado.
# El 'push --force' se usó para sobrescribir el historial debido a una corrección.
